FUNCTION  craftIronIngot() 
    
    IF inventoryContains("IRON ORE",  3)  THEN
        
	REMOVE 3 "IRON ORE" FROM INVENTORY
        ADD "IRON INGOT" TO CraftedItem
        PRINT "Crafted Iron Ingot."
    
    ELSE 
        PRINT "Insufficient resources to craft Iron Ingot."

    ENDIF

END FUNCTION
