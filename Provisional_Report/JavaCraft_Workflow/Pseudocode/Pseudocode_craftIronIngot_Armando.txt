PSEUDOCODE craftIronIngot()

FUNCTION  craftIronIngot() 

BEGIN

    IF inventory CONTAINS 3 "IRON ORE" THEN

	        REMOVE 3 "IRON ORE" FROM INVENTORY
            ADD "IRON INGOT" TO CraftedItem
            OUTPUT "Crafted Iron Ingot."
    
    ELSE 
        OUTPUT "Insufficient resources to craft Iron Ingot."

    ENDIF

END FUNCTION
