Pseudocode for the game

BEGIN
    SETUP world
    SETUP player
    PRINT welcome message and instructions
    INPUT startGameChoice
    IF startGameChoice EQUALS "Y" THEN
        INITIALIZE secret door flags
        WHILE TRUE
            CLEAR screen
            PRINT user interface // legend, world, inventory, action menu
            INPUT actionChoice

            IF actionChoice EQUALS "EXIT" THEN
                BREAK // breaks out of the while loop and ends the game
            END IF

            SWITCH actionChoice
                CASE MOVE:
                CASE MINE:
                    IF unlockMode THEN UPDATE secret door flags
                    END IF
                    EXECUTE actionChoice
                    BREAK
                CASE UNLOCK:
                    SET unlockMode to TRUE
                    BREAK
                CASE OPEN:
                    IF unlockMode && (all secret door flags) THEN
                        SET secretDoorUnlocked to TRUE
                        RESET world
                        PRINT message
                    ELSE
                        SET (all secret door flags) to FALSE
                        SET unlockMode to FALSE
                    END IF
                    BREAK
                DEFAULT: 
                    EXECUTE actionChoice

            IF unlockMode THEN
                UPDATE secret door flags // check if player's action changes the state of any flag
            END IF

            IF secretDoorUnlocked THEN 
                CLEAR screen
                PRINT message
                RESET world // Set world to secret area
                FILL inventory
            END IF

        END WHILE
    ELSE
    PRINT goodbye message
    END IF
END
