FUNCTION getCraftedItemName(craftedItem):
    IF craftedItem IS EQUAL TO CRAFTED_WOODEN_PLANKS:
        return "Wooden Planks"
    ELSE IF craftedItem IS equal TO CRAFTED_STICK:
        return "Stick"
    ELSE IF craftedItem IS EQUAL TO CRAFTED_IRON_INGOT:
        return "Iron Ingot"
    ELSE IF craftedItem IS EQUAL TO CRAFTED_GOLD_RING:
        RETURN "Gold Ring"
    ELSE:
        RETURN "Unknown"
END FUNCTION
