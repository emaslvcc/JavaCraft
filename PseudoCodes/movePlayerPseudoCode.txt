Take 1 parameter: String direction

IF direction.toUpperCase == "W" OR "UP"
	IF playerY > 0
		playerY--

ELIF direction.toUpperCase == "S" OR "DOWN"
	IF playerY < worldHeight - 1
		playerY++

ELIF direction.toUpperCase == "A" OR "LEFT"
	IF playerX > 0
		playerX--

ELIF direction.toUpperCase == "D" OR "RIGHT"
	IF playerX < worldWidth -1
		playerX++
