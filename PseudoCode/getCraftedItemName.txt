function getCraftedItemName

   Switch blocktype

	case blocktype has the same value as the item id of wooden planks
		return the string “Wooden Planks”
	 
	case blocktype has the same value as the item id of stick
		return the string “Stick”

	case blocktype has the same value as the id of iron ingots
		return the string “Iron Ingots”
	default
 	  	return -1 
     END Switch
