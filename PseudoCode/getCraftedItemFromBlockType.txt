Function getCraftedItemFromBlockType

   Switch blocktype
	case  blockType is equal (id of wooden planks)
		return CRAFTED_WOODEN_PLANKS
	case  blockType is (id of stick)
		return CRAFTED_STICK
	case  blockType is (id of iron ingots)
		return CRAFTED_IRON_INGOT
	default
 	 	return -1
     END Switch
