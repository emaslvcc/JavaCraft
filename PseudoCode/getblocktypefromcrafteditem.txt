Function getBlockTypeFromCraftedItem
   Switch craftedItem
	case CraftedItem is equal to the integer CRAFTED_WOODEN_PLANKS
		return 5
	case CraftedItem is equal to the integer CRAFTED_STICK
		return 6
	case CraftedItem is equal to the integer CRAFTED_IRON_INGOT
		return 7
	default
 	 	return -1
     END 	 Switch
