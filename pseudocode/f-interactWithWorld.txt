function interactWithWorld()

blockType is the block the player is currently standing on

switch(blockType):

case WOOD: 
	print("You gather wood from the tree.")
	add 1 wood to inventory
end if

case LEAVES:
	print("You gather leaves from the tree.")
	add 1 leaves item to inventory
end if

case STONE:
	print("You gather stones from the ground.")
	add 1 stone to inventory
end if

case IRON_ORE:
	print("You mine iron ore from the ground.")
	add 1 iron ore to inventory
end if

case AIR:
	print("Nothing to interact with here.")
end if

default: print("Unrecognized block. Cannot interact.")

end switch

wait for enter

end function
